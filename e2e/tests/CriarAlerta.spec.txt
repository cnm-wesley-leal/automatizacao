import { Browser, Locator, Page, expect, test } from '@playwright/test'
import { REALTY_SEARCH_DATA } from '../utils/test-data'
import { dismissCookieConsent } from '../utils/helpers'

const D = REALTY_SEARCH_DATA

type CriarAlertaHealth = {
  managementRouteAvailable: boolean
  managementRouteReason: string
  createFlowAvailable: boolean
  createFlowReason: string
}

let cachedCriarAlertaHealth: Promise<CriarAlertaHealth> | null = null

async function resolveCriarAlertaHealth(browser: Browser): Promise<CriarAlertaHealth> {
  if (cachedCriarAlertaHealth) return cachedCriarAlertaHealth

  cachedCriarAlertaHealth = (async () => {
    const context = await browser.newContext({ baseURL: process.env.BASE_URL })
    const page = await context.newPage()

    try {
      const managementResponse = await page.request.get('/alertas-de-busca/', { failOnStatusCode: false })
      const managementStatus = managementResponse.status()
      const managementRouteAvailable = managementStatus < 500

      let createFlowAvailable = false
      let createFlowReason = 'Fluxo de criação indisponível para automação no ambiente atual.'

      try {
        await gotoRealtyListing(page, true)
        const trigger = await createAlertTrigger(page)
        await expect(trigger).toBeVisible({ timeout: 5000 })

        if (!(await trigger.isEnabled())) {
          createFlowReason = 'Gatilho de criação está visível porém desabilitado no ambiente atual.'
        } else {
          await trigger.click()

          const redirectedToAuth = await page
            .waitForURL(/login|entrar|acessar/i, { timeout: 3000 })
            .then(() => true)
            .catch(() => false)

          const hasDialog = await page.getByRole('dialog').first().isVisible().catch(() => false)
          const hasContinueButton = await page
            .getByRole('button', { name: /continuar/i })
            .first()
            .isVisible()
            .catch(() => false)

          createFlowAvailable = redirectedToAuth || hasDialog || hasContinueButton
          if (!createFlowAvailable) {
            createFlowReason =
              'Não foi possível detectar superfície do fluxo de criação (login/modal/continuar) após clique em Criar alerta.'
          }
        }
      } catch (error) {
        createFlowReason = `Falha no health check do fluxo de criação: ${String(error)}`
      }

      return {
        managementRouteAvailable,
        managementRouteReason: managementRouteAvailable
          ? 'Rota /alertas-de-busca saudável para execução automatizada.'
          : `Rota /alertas-de-busca indisponível (status ${managementStatus}).`,
        createFlowAvailable,
        createFlowReason,
      }
    } finally {
      await context.close()
    }
  })()

  return cachedCriarAlertaHealth
}

async function createAlertTrigger(page: import('@playwright/test').Page) {
  const button = page.getByRole('button', { name: /criar alerta/i }).first()
  if (await button.count()) return button
  return page.getByRole('link', { name: /criar alerta/i }).first()
}

async function gotoRealtyListing(page: import('@playwright/test').Page, withRequiredFilters = true) {
  const url = withRequiredFilters
    ? '/apartamentos-a-venda/pr-curitiba/'
    : '/imoveis/brasil/'
  await page.goto(url, { waitUntil: 'domcontentloaded' })
  await dismissCookieConsent(page)
}

async function assertBlockedCreateAction(page: import('@playwright/test').Page) {
  const requests: string[] = []
  const onRequest = (req: import('@playwright/test').Request) => {
    if (req.url().includes('/search-alert')) requests.push(req.url())
  }
  page.on('request', onRequest)

  const trigger = await createAlertTrigger(page)
  await expect(trigger).toBeVisible()

  if (await trigger.isEnabled()) {
    await trigger.click()
  }

  await expect(page).toHaveURL(/qa\.chavesnamao\.com\.br/)
  expect(requests.length, 'Fluxo sem critérios obrigatórios não deve disparar request de criação').toBe(0)

  page.off('request', onRequest)
}

async function gotoAlertManagement(page: Page) {
  await page.goto('/alertas-de-busca/', { waitUntil: 'domcontentloaded' })
  await dismissCookieConsent(page)
  await expect(page).toHaveURL(/alertas-de-busca/i)
  await expect(page.getByRole('heading', { name: /alertas de busca/i })).toBeVisible()
}

async function getVisibleMainAction(page: Page, pattern: RegExp): Promise<Locator | null> {
  const main = page.locator('main').first()
  const button = main.getByRole('button', { name: pattern }).first()
  if (await button.isVisible().catch(() => false)) return button

  const link = main.getByRole('link', { name: pattern }).first()
  if (await link.isVisible().catch(() => false)) return link

  return null
}

async function getVisibleScopedAction(scope: Locator, pattern: RegExp): Promise<Locator | null> {
  const button = scope.getByRole('button', { name: pattern }).first()
  if (await button.isVisible().catch(() => false)) return button

  const link = scope.getByRole('link', { name: pattern }).first()
  if (await link.isVisible().catch(() => false)) return link

  return null
}

async function getAlertManagementSignals(page: Page) {
  const emptyStateCTA = page.getByRole('button', { name: /buscar seu pr[oó]ximo im[oó]vel/i }).first()
  const hasEmptyState = await emptyStateCTA.isVisible().catch(() => false)

  const resultsAction = await getVisibleMainAction(page, /ver resultados|resultados de busca/i)
  const detailsAction = await getVisibleMainAction(page, /par[aâ]metros de busca|ver par[aâ]metros|detalhes/i)
  const deleteAction = await getVisibleMainAction(page, /excluir|lixeira|remover alerta/i)
  const clearAllAction = await getVisibleMainAction(page, /limpar alertas|excluir todos|apagar todos/i)

  return {
    hasEmptyState,
    emptyStateCTA,
    resultsAction,
    detailsAction,
    deleteAction,
    clearAllAction,
  }
}

async function countVisibleDeleteActions(page: Page) {
  const main = page.locator('main').first()
  const buttonCount = await main
    .getByRole('button', { name: /excluir|lixeira|remover alerta/i })
    .filter({ visible: true })
    .count()
  const linkCount = await main
    .getByRole('link', { name: /excluir|lixeira|remover alerta/i })
    .filter({ visible: true })
    .count()
  return buttonCount + linkCount
}

async function openDeletionConfirmation(page: Page) {
  const deleteAction = await getVisibleMainAction(page, /excluir|lixeira|remover alerta/i)
  test.skip(!deleteAction, 'Sem ação de exclusão disponível para validar este cenário no usuário atual.')

  await deleteAction!.click()

  const modal = page.getByRole('dialog').first()
  const modalVisible = await modal.isVisible().catch(() => false)
  if (modalVisible) {
    await expect(modal).toContainText(/excluir|remover|tem certeza|confirmar/i)
    return modal
  }

  await expect(page.locator('main')).toContainText(/excluir|remover|tem certeza|confirmar/i)
  return page.locator('main')
}

test.describe('Criar Alerta — Cenários Automatizados (M1, M2, M3, M4, M4B, M6)', () => {
  test('CT01 - usuário logado com filtros visualiza botão Criar Alerta', async ({ page }) => {
    await gotoRealtyListing(page, true)
    const trigger = await createAlertTrigger(page)
    await expect(trigger).toBeVisible()
  })

  test('CT02 - usuário não logado clica em Criar Alerta e deve ir para login', async ({ browser }) => {
    const context = await browser.newContext({ baseURL: process.env.BASE_URL })
    const page = await context.newPage()
    await gotoRealtyListing(page, true)

    const trigger = await createAlertTrigger(page)
    await expect(trigger).toBeVisible()

    if (await trigger.isEnabled()) {
      await trigger.click()
      const redirectedToLogin = await page
        .waitForURL(/login|entrar|acessar/i, { timeout: 3000 })
        .then(() => true)
        .catch(() => false)

      if (!redirectedToLogin) {
        await expect(page).toHaveURL(/apartamentos-a-venda\/pr-curitiba/i)
        await expect(page.getByRole('link', { name: /entrar/i })).toBeVisible()
      }
    } else {
      await expect(page.getByRole('link', { name: /entrar/i })).toBeVisible()
    }
    await context.close()
  })

  test('CT03 - sem filtros obrigatórios não deve permitir criação', async ({ page }) => {
    await gotoRealtyListing(page, false)
    const trigger = await createAlertTrigger(page)

    if (await trigger.isVisible().catch(() => false)) {
      if (await trigger.isDisabled()) {
        await expect(trigger).toBeDisabled()
        return
      }

      await trigger.click()
      await expect(page).toHaveURL(/imoveis\/brasil|login|entrar/i)
      return
    }

    await expect(trigger).toHaveCount(0)
  })

  test('CT04 - em contexto de poucos/zero resultados deve exibir CTA para criar alerta', async ({ page }) => {
    await page.goto(D.urls.zeroResults, { waitUntil: 'domcontentloaded' })

  test('CT05 - bloquear criação sem finalidade selecionada', async ({ page }) => {

    const dialogVisible = await dialog.isVisible().catch(() => false)
    const detailsScope = dialogVisible ? dialog : page.locator('main')

    await expect(detailsScope).toContainText(/par[aâ]metros|crit[ée]rios|detalhes/i)

  test('CT21 - excluir alerta individual com confirmação', async ({ browser, page }) => {
    const health = await resolveCriarAlertaHealth(browser)
    test.skip(!health.managementRouteAvailable, `CT21 bloqueado por health check: ${health.managementRouteReason}`)

    await gotoAlertManagement(page)
    const signals = await getAlertManagementSignals(page)
    test.skip(signals.hasEmptyState, 'Sem alertas ativos para validar exclusão individual (CT21).')

    const beforeCount = await countVisibleDeleteActions(page)
    test.skip(beforeCount < 1, 'Sem mais de um alerta com ação de exclusão visível para validar CT21.')

    const confirmationScope = await openDeletionConfirmation(page)
    const confirmAction =
      (await getVisibleMainAction(page, /sim|confirmar|excluir/i)) ||
      (await getVisibleScopedAction(confirmationScope, /sim|confirmar|excluir/i))

    test.skip(!confirmAction, 'Modal de confirmação sem ação explícita de confirmar exclusão para CT21.')
    await confirmAction!.click()

    await expect(page.getByRole('dialog').first()).not.toBeVisible({ timeout: 5000 }).catch(() => {})
    const afterCount = await countVisibleDeleteActions(page)
    const hasEmptyStateAfterDelete = await page
      .getByRole('button', { name: /buscar seu pr[oó]ximo im[oó]vel/i })
      .first()
      .isVisible()
      .catch(() => false)

    expect(
      afterCount < beforeCount || hasEmptyStateAfterDelete,
      'Após confirmação de exclusão individual, a lista deve reduzir ou entrar em estado vazio.'
    ).toBe(true)
  })

  test('CT22 - cancelar exclusão de alerta', async ({ browser, page }) => {
    const health = await resolveCriarAlertaHealth(browser)
    test.skip(!health.managementRouteAvailable, `CT22 bloqueado por health check: ${health.managementRouteReason}`)

    await gotoAlertManagement(page)

    test.skip(signals.hasEmptyState, 'Sem alertas ativos para validar cancelamento de exclusão (CT22).')

    const beforeCount = await countVisibleDeleteActions(page)
    test.skip(beforeCount < 1, 'Sem alerta com ação de exclusão disponível para validar CT22.')

    const confirmationScope = await openDeletionConfirmation(page)
    const cancelAction = await getVisibleScopedAction(confirmationScope, /n[aã]o|nao|cancelar|fechar/i)

    test.skip(!cancelAction, 'Modal de confirmação sem ação explícita de cancelamento para CT22.')
    await cancelAction!.click()

    await expect(page.getByRole('dialog').first()).not.toBeVisible({ timeout: 5000 }).catch(() => {})
    const afterCount = await countVisibleDeleteActions(page)
    expect(afterCount, 'Ao cancelar exclusão, a lista deve permanecer inalterada.').toBe(beforeCount)

    expect(title).not.toMatch(/not found/i)

  })
